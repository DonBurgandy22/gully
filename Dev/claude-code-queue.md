    def send_message(self, text):

            "type": "message",
            "text": text,
            "session_id": self.session_id,
            "channel": "voice"
        })
    
    def receive_messages(self):
        """Receive messages from OpenClaw"""
        for message in self.ws.listen():
            if message["type"] == "response":
                return message["text"]
            elif message["type"] == "tool_result":
